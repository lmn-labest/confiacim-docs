class Dist:
    """
    Classe base para as distribuições.

    Parameters:
        dist: Será atribuido a uma função do `scipy`.

    Danger:
        Esta classe não deve ser usada diretamente.
    """

    dist = None

    def __init__(self, mu: float, delta: float):
        """
        Inicializador da classe.

        Parameters:
            mu: média.
            delta: coeficiente de variação.

        Info:
            coeficiente de variação é o desvio padrão divido pela média.
        """
        self.mu = mu
        self.delta = delta

    def _params_dist(self):
        """Esta função serve para ajutar os paramentros necessarios paras as diferentes distribuição o scipy"""
        return {"loc": self.loc, "scale": self.scale}

    @property
    def sigma(self) -> float:
        """Retorna do desvio padrão"""
        return self.mu * self.delta

    @property
    def pdf(self):
        """Função densidade de probabilidade da distruibuição `(pdf)`"""
        return partial(self.dist.pdf, **self._params_dist())

    @property
    def cdf(self):
        """Função densidade de probabilidade acumulada da distruibuição `(cdf)`"""
        return partial(self.dist.cdf, **self._params_dist())

    @property
    def ppf(self):
        """Inversa da função densidade de probabilidade acumulada da distruibuição `(cdf)`"""
        return partial(self.dist.ppf, **self._params_dist())

class DistGumbelR(Dist):
    """
    Classe da distruibução `gumbel_r`.

    Parameters:
        dist: função `gumbel_r` do `scipy`.
    """

    dist = gumbel_r

    def __init__(self, mu: float, delta: float):
        """
        Inicializador da classe.

        Parameters:
            mu: média
            delta: coeficiente de variação.
        """
        super().__init__(mu, delta)
        alpha, u = self._paramenters()
        self.loc = u
        self.scale = 1.0 / alpha

    def _paramenters(self) -> tuple[float, float]:
        """
        Returns:
            Returna uma tupla com a média, ksi e o lambda.
        """

        def alpha(sig: float) -> float:
            return (pi / sqrt(6)) * (1.0 / sig)

        def u(mu: float, alpha: float) -> float:
            return mu - 0.5772 / alpha

        alpha_ = alpha(self.sigma)
        return alpha_, u(self.mu, alpha_)

class DistLogNormal(Dist):
    """
    Classe da distruibução `lognormal`.

    Parameters:
        dist: função `lognorm` do `scipy`.
    """

    dist = lognorm

    def __init__(self, mu: float, delta: float):
        """
        Inicializador da classe.

        Parameters:
            mu: média
            delta: coeficiente de variação
        """
        super().__init__(mu, delta)
        _, ksi, lambda_ = self._paramenters()
        self.s = ksi
        self.scale = exp(lambda_)

    def _paramenters(self) -> tuple[float, float, float]:
        """
        Returns:
            Returna uma tupla com a média, ksi e o lambda.
        """

        def ksi(delta: float) -> float:
            return sqrt(log(1.0 + delta**2))

        def lambda_(mu: float, ksi: float) -> float:
            return log(mu) - 0.5 * ksi**2

        k = ksi(self.delta)
        return self.sigma, k, lambda_(self.mu, k)

    def _params_dist(self):
        """Esta função serve para ajutar os paramentros necessarios paras as diferentes distribuição o scipy"""
        return {"s": self.s, "scale": self.scale}

class DistNormal(Dist):
    """
    Classe da distruibução `normal`.

    Parameters:
        dist: função `norm` do `scipy`.
    """

    dist = norm

    def __init__(self, mu: float, delta: float):
        """
        Inicializador da classe.

        Parameters:
            mu: média.
            delta: coeficiente de variação.
        """

        super().__init__(mu, delta)
        _, sig = self._paramenters()
        self.loc = self.mu
        self.scale = sig

    def _paramenters(self) -> tuple[float, float]:
        """
        Returns:
            Returna uma tupla com a média e o desvio padrão.
        """
        return self.mu, self.sigma

class DistributionNames(Enum):
    """
    Ditruibuição disponiveis.

    Parameters:
        NORMAL: distruibuição normal.
        LOGNORMAL: distruibuição lognormal.
        GUMBEL_R: distruibuição de gumbel com cauda a direita.

    """

    NORMAL = "normal"
    LOGNORMAL = "lognormal"
    GUMBEL_R = "gumbel_r"

    @classmethod
    def has_value(cls, value):
        return value.upper() in cls._member_names_

    @classmethod
    def options(cls):
        return [d.name.lower() for d in cls]

def set_distributions(variables: tuple[StochasticVariable, ...]) -> dict[str, list]:
    """
    Cria uma dicionário com as respectivas distribuicões. As médias são sempre unitárias.

    Parameters:
        variables: Variáves aleatórias.

    Returns:
        Retorna um dicionário com as distruibuições com sua respectivamente `pdf`, `cdf` e `ppf`.
    """
    fs, Fs, IFs = [], [], []

    dist_options = {
        "lognormal": DistLogNormal,
        "normal": DistNormal,
        "gumbel_r": DistGumbelR,
    }

    for v in variables:
        try:
            Dist = dist_options[v.dist]
        except KeyError:
            raise ValueError(f"Distribuição {v.dist} da variável {v.name} é invalida.") from KeyError

        # Internamente das distribuições são sempre tradas como unitárias
        dist = Dist(1.0, v.cov)
        fs.append(dist.pdf)
        Fs.append(dist.cdf)
        IFs.append(dist.ppf)

    return dict(fs=fs, Fs=Fs, IFs=IFs)
