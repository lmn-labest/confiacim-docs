def coefficient_must_have_abs_value_of_one(
    correlations: dict[str, float],
    raise_exception: bool = True,
) -> bool:
    """Verifica se os valores da correlação estão entre -1.0 e 1.0.

    Parameters:
        correlations: dicionário com as correlações entre as variáveis.
        raise_exception: Se True, levanta uma exceção se o valor não estiver entre -1.0 e 1.0.

    Raises:
        CorrelationValueError: O valor precisa estar entre -1.0 e 1.0.

    Returns:
        Retorna True se todos os valores entre -1.0 e 1.0, caso contrário, retorna False..
    """

    for k, v in correlations.items():
        if not -1.0 < v < 1.0:
            if not raise_exception:
                return False
            raise CorrelationValueError(k, v)

    return True

def generate_correlation_matrix(
    correlations: dict[str, float] | None, variables: tuple[StochasticVariable, ...]
) -> np.ndarray:
    """Calcula a matrix de correlação.

    Parameters:
        correlations: dicionário com as correlações entre as variáveis.
        variables: variáveis aleatorios.

    Returns:
        Retorna a matriz de correlações
    """
    n_vars = len(variables)
    ro = np.identity(n_vars, dtype=np.float64)

    if not correlations:
        return ro

    coefficient_must_have_abs_value_of_one(correlations)

    index = {}
    for i, v in enumerate(variables):
        index[v.name] = i

    for k, coor in correlations.items():
        name1, name2 = map(str.strip, k.split(","))
        i, j = index[name1], index[name2]
        ro[i, j] = ro[j, i] = coor

    return ro

def matrix_is_spd(matrix: np.ndarray) -> bool:
    """Verifica se a matriz é simétrica positiva definida.

    Parameters:
        matrix: matriz a ser verificada.

    Returns:
        Retorna True se a matriz for simétrica positiva definida, caso contrário, retorna False.
    """
    try:
        np.linalg.cholesky(matrix)
    except np.linalg.LinAlgError:
        return False
    return True
