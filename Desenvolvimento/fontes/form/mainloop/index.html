def loop(
    input_dir: Path,
    output_dir: Path,
    variables: tuple[FormVariabel, ...],
    beta: float,
    delta: float,
    gamma: float,
    tol: float = 1.0e-05,
    maxit: int = 100,
    correlations: dict[str, float] | None = None,
    verbose: bool = True,
    console: Console | None = None,
) -> FormResult:
    """
    Efetua os cálculos e procedientos do loop princial do método **FORM**.

    Parameters:
        input_dir: Diretório dos arquivos de entrada.
        output_dir: Diretório dos arquivos de saida.
        variables: Variaveis aleatórias.
        beta: Parametro do **FORM**.
        delta: Parametro do **FORM**.
        gamma: Parametro do **FORM**.
        tol: Tolerância do **FORM**.
        maxit: Número máximo de iterações do **FORM**.
        correlations: Informação das correlações das variáveis.
        verbose: verbosidade da funçãao no console.
        console: console do `Rich`.

    Returns:
        Retorna o resultado final do **FORM**.
    """

    distributions = set_distributions(variables)

    n_vars = len(variables)

    Ua = np.ones(n_vars, dtype=np.float64)

    ro = generate_correlation_matrix(correlations, variables)

    L = cholesky(ro, lower=True)

    avgs = sample_averages(variables)

    res = 1.0

    with Progress(
        SpinnerColumn(),
        TextColumn("[progress.description]{task.description}\n"),
        transient=True,
        disable=verbose,
    ) as progress:
        task = progress.add_task(description="Running...", total=None)

        # --- start mainloop
        for i in range(maxit):
            progress.update(
                task,
                description=(
                    f"[green]Running[/green]:\n"
                    f"[cyan]Iteration[/cyan] = {i+1:8}\n"
                    f"[cyan]Residual[/cyan]  = {res:.2e}"
                ),
            )
            if verbose and console:
                text = Text(f"Iteration {i + 1}", justify="center", style=COLORS[0])
                panel = Panel(Align(text, vertical="middle", align="center"))
                console.print(panel)

                text = Text("Temcim step", justify="center", style=COLORS[1])
                panel = Panel(Align(text, vertical="middle", align="center"))
                console.print(panel)

            # Tencim
            generator = SamplesGenerator(avgs, delta=delta)
            samples = generator.all()

            rcs = run_samples_iteration_i(
                samples=samples,
                input_dir=input_dir,
                output_dir=output_dir,
                iteration=i + 1,
                verbose=verbose,
            )

            write_samples_variables_it(samples, i + 1, output_dir)

            # form
            if verbose and console:
                text = Text("End temcim step", justify="center", style=COLORS[1])
                panel = Panel(Align(text, vertical="middle", align="center"))
                console.print(panel)

                text = Text("form step", justify="center", style=COLORS[2])
                panel = Panel(Align(text, vertical="middle", align="center"))
                console.print(panel)

                console.print(tables.variables_vector(Ua, avgs, f"U_{i + 1}"))

                console.print(tables.samples_it_rcs(rcs))

            Ua, betai, res, alpha = form_step(Ua, L, rcs, gamma, delta, beta, distributions)

            avgs = Ua2avgs_denormalized(Ua, variables)

            write_variables_it(avgs, i + 1, output_dir)

            write_iteration_log(betai, res, i + 1, output_dir)

            if verbose and console:
                console.print(tables.convergence(betai, res, "Convergence"))

                text = Text("End form step", justify="center", style=COLORS[2])
                panel = Panel(Align(text, vertical="middle", align="center"))
                console.print(panel)

                console.print(tables.variables_vector(Ua, avgs, "U_new"))

                text = Text(f"End iteration {i + 1}", justify="center", style=COLORS[0])
                panel = Panel(Align(text, vertical="middle", align="center"))
                console.print(panel)

            beta = betai

            if res < tol:
                break

    # --- end mainloop
    result = FormResult(
        beta=beta,
        res=res,
        rcs=rcs,
        it=i + 1,
        Pf=norm.cdf(-beta),
        importance_factors=importance_factors(alpha),
        omission_factors=omission_factors(alpha),
    )

    return result
